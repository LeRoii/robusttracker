cmake_minimum_required(VERSION 3.4.1)

project(rknn_yolov8_demo)

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# skip 3rd-party lib dependencies
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} -Wl,--allow-shlib-undefined")

# Set the install prefix to /usr instead of the default /usr/local
set(CMAKE_INSTALL_PREFIX /usr)

set(CMAKE_SKIP_INSTALL_RPATH FALSE)
set(CMAKE_BUILD_WITH_INSTALL_RPATH TRUE)
set(CMAKE_INSTALL_RPATH "${CMAKE_INSTALL_PREFIX}/lib")

# opencv
find_package(OpenCV REQUIRED)

# rga
include_directories(/usr/include/rga)

# rknn_yolov8_demo
include_directories(${CMAKE_SOURCE_DIR}/include)

# 创建动态库而不是可执行文件
add_library(idetector_yolov8 SHARED
    src/model_inference.cpp
    src/idetector.cpp
)

target_link_libraries(idetector_yolov8
  rknn_api
  ${OpenCV_LIBS}
  pthread
)

# 指定库文件的安装路径为/usr/lib
install(TARGETS idetector_yolov8 LIBRARY DESTINATION lib)
